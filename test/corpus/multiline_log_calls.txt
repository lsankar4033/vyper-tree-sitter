    strategy: address
    gain: uint256
    loss: uint256
    current_debt: uint256
    protocol_fees: uint256
    total_fees: uint256
    total_refunds: uint256

@external
def emit_events():
    log UpdateProtocolFeeBps(
        old_fee,
        new_fee
    )

    log StrategyReported(
        strategy,
        gain,
        loss,
        current_debt,
        total_fees * convert(protocol_fee_bps, uint256) / 10_000, # Protocol Fees
        total_fees,
        total_refunds
    )
